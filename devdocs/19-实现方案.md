# 19 — naiz_img 实现方案（到 M1 蓝屏）

## 概述

按依赖顺序，从底向上实现 `tools/naiz_img/`，最终用 `inject.py` 产出 HDI，在 NP2kai 上验证蓝屏。

## 实施步骤

### Step 1：`base.py` — DiskImage 基类

**文件**：`tools/naiz_img/base.py`

实现 `DiskImage` 抽象基类。所有容器格式共享此接口。

```python
class DiskImage:
    def __init__(self, path):
        with open(path, 'rb') as f:
            self._data = bytearray(f.read())
        self._parse()

    def _parse(self):
        raise NotImplementedError

    @property
    def sector_size(self): return self._sector_size
    @property
    def total_sectors(self): return self._total_sectors

    def read_sector(self, lba) -> bytes:
        off = lba * self._sector_size
        return bytes(self._data[off:off + self._sector_size])

    def read_sectors(self, lba, count) -> bytes:
        out = bytearray()
        for i in range(count):
            out.extend(self.read_sector(lba + i))
        return bytes(out)

    def write_sector(self, lba, data):
        off = lba * self._sector_size
        end = off + self._sector_size
        self._data[off:end] = data[:self._sector_size]

    def save(self, path=None):
        dst = path or self.path
        with open(dst, 'wb') as f:
            f.write(self._data)
```

**验证**：手动构造子类实例，调用 `read_sector(0)` 返回非空数据即可。

### Step 2：`hdi.py` — HDIImage

**文件**：`tools/naiz_img/hdi.py`

解析 msdos5.hdi 的 4096 字节头部。直接支持现有镜像是下一步所有工作的前提。

HDI 头部布局（参考 Anex86 规范）：

| 偏移 | 类型 | 字段 |
|------|------|------|
| 0x00 | u32 | reserved |
| 0x04 | u32 | hdd_type |
| 0x08 | u32 | hdr_size（= 4096） |
| 0x0C | u32 | data_size |
| 0x10 | u32 | sector_size |
| 0x14 | u32 | spt |
| 0x18 | u32 | heads |
| 0x1C | u32 | cylinders |

```python
class HDIImage(DiskImage):
    def _parse(self):
        hdr_size = struct.unpack_from('<I', self._data, 0x08)[0]
        sec_size = struct.unpack_from('<I', self._data, 0x10)[0]
        spt      = struct.unpack_from('<I', self._data, 0x14)[0]
        heads    = struct.unpack_from('<I', self._data, 0x18)[0]
        cyls     = struct.unpack_from('<I', self._data, 0x1C)[0]
        self._sector_size = sec_size
        self._raw_offset  = hdr_size
        self._total_sectors = (len(self._data) - hdr_size) // sec_size
        self._spt = spt
        self._heads = heads
        self._cyls = cyls

    def read_sector(self, lba):
        off = self._raw_offset + lba * self._sector_size
        return bytes(self._data[off:off + self._sector_size])

    def write_sector(self, lba, data):
        off = self._raw_offset + lba * self._sector_size
        self._data[off:off + self._sector_size] = data[:self._sector_size]
```

**验证**：

```python

assert img.sector_size == 512
assert img.total_sectors == 722 * 8 * 17
s0 = img.read_sector(0)
assert s0[3:7] == b'IPL1'
```

### Step 3：`partition.py` — 分区表检测

**文件**：`tools/naiz_img/partition.py`

检测 PC-98 IPL 分区表，定位 FAT 分区起始位置。

```python
@dataclass
class PartitionEntry:
    scheme: str        # "MBR" 或 "PC-98"
    type_id: int
    byte_offset: int
    byte_size: int

def detect_mbr(img) -> list[PartitionEntry]:
    sec0 = img.read_sector(0)
    if sec0[0x1FE:0x200] != b'\x55\xAA':
        return []
    parts = []
    for i in range(4):
        off = 0x1BE + i * 16
        typ = sec0[off + 4]
        if typ == 0: continue
        lba = struct.unpack_from('<I', sec0, off + 8)[0]
        size = struct.unpack_from('<I', sec0, off + 12)[0]
        parts.append(PartitionEntry("MBR", typ, lba * img.sector_size, size * img.sector_size))
    return parts

def detect_pc98(img) -> list[PartitionEntry]:
    sec0 = img.read_sector(0)
    if sec0[3:7] != b'IPL1':
        return []
    sec1 = img.read_sector(1)
    spt = getattr(img, '_spt', 17)
    heads = getattr(img, '_heads', 8)
    parts = []
    for i in range(16):
        off = i * 32
        entry = sec1[off:off + 32]
        if all(b == 0 for b in entry):
            continue
        sys_id = entry[1]
        start_head = entry[8]
        start_sec = entry[9]
        start_cyl = struct.unpack_from('<H', entry, 10)[0]
        end_head = entry[12]
        end_sec = entry[13]
        end_cyl = struct.unpack_from('<H', entry, 14)[0]
        lba_start = start_cyl * heads * spt + start_head * spt + (start_sec - 1)
        lba_end = end_cyl * heads * spt + end_head * spt + end_sec
        byte_off = lba_start * img.sector_size
        byte_size = (lba_end - lba_start) * img.sector_size
        parts.append(PartitionEntry("PC-98", sys_id, byte_off, byte_size))
    return parts

def detect_partitions(img) -> list[PartitionEntry]:
    parts = detect_mbr(img)
    if parts: return parts
    return detect_pc98(img)
```

**对 msdos5.hdi 的预期**：
- `detect_pc98` 在扇区 1 找到分区，start_cyl=1, head=0, sector=1 → LBA=136
- 返回 `PartitionEntry(byte_offset=136*512=69632, ...)`

**验证**：

```python

assert len(parts) > 0

vbr = img.read_sector(vbr_lba)

assert bps in (512, 1024)
```

### Step 4：`fat.py` — FAT12/FAT16 文件系统

**文件**：`tools/naiz_img/fat.py`

最核心的模块。分为两大部分：**解析**（读 FAT + 目录树）和 **写入**（write_back）。

#### 4.1 数据结构

```python
class FileEntry:
    __slots__ = ('name', 'ext', 'attr', 'cluster', 'size', 'children')
    def __init__(self, name, ext, attr, cluster, size):
        self.name = name.strip()
        self.ext = ext.strip()
        self.attr = attr
        self.cluster = cluster
        self.size = size
        self.children = {}

    @property
    def is_directory(self): return bool(self.attr & 0x10)
    @property
    def display_name(self): return self.name if not self.ext else f"{self.name}.{self.ext}"
```

#### 4.2 BPB 解析与布局计算

```python
ATTR_DIRECTORY = 0x10
ATTR_ARCHIVE   = 0x20
ATTR_SYSTEM    = 0x04

FAT12_EOC = 0xFF8
FAT16_EOC = 0xFFF8

class NAIZFatFS:
    def __init__(self, img, part_offset=None):
        self.img = img
        self.part_offset = part_offset

        if part_offset is None:
            parts = detect_partitions(img)
            if not parts:
                raise ValueError("No partition found")
            self.part_offset = parts[0].byte_offset

        vbr_lba = self.part_offset // img.sector_size
        vbr = img.read_sector(vbr_lba)

        self.bytes_per_sector = struct.unpack_from('<H', vbr, 0x0B)[0]
        self.sectors_per_cluster = vbr[0x0D]
        self.reserved_sectors = struct.unpack_from('<H', vbr, 0x0E)[0]
        self.num_fats = vbr[0x10]
        self.root_entries = struct.unpack_from('<H', vbr, 0x11)[0]
        self.fat_sectors = struct.unpack_from('<H', vbr, 0x16)[0]
        self.media_descriptor = vbr[0x15]
        total_16 = struct.unpack_from('<H', vbr, 0x13)[0]
        total_32 = struct.unpack_from('<I', vbr, 0x20)[0]
        self.total_sectors = total_16 if total_16 else total_32

        self.root_sectors = (self.root_entries * 32 + self.bytes_per_sector - 1) // self.bytes_per_sector
        self.fat_offset = self.part_offset + self.reserved_sectors * self.bytes_per_sector
        self.root_offset = self.fat_offset + self.num_fats * self.fat_sectors * self.bytes_per_sector
        self.data_offset = self.root_offset + self.root_sectors * self.bytes_per_sector
        self.cluster_size = self.bytes_per_sector * self.sectors_per_cluster

        data_sectors = self.total_sectors - self.reserved_sectors - self.num_fats * self.fat_sectors - self.root_sectors
        total_clusters = data_sectors // self.sectors_per_cluster
        self.fat_type = 12 if total_clusters < 4085 else 16
        self._fat_data = self._read_bytes(self.fat_offset, self.fat_sectors * self.bytes_per_sector)
        self._build_tree()
```

#### 4.3 FAT 表读写

```python
def _read_fat_entry(self, cluster):
    if self.fat_type == 12:
        offset = cluster + (cluster // 2)
        word = struct.unpack_from('<H', self._fat_data, offset)[0]
        return (word >> 4) if (cluster & 1) else (word & 0x0FFF)
    else:
        return struct.unpack_from('<H', self._fat_data, cluster * 2)[0]

def _get_cluster_chain(self, start):
    chain = []
    c = start
    eoc = FAT12_EOC if self.fat_type == 12 else FAT16_EOC
    while c >= 2 and c < eoc:
        chain.append(c)
        c = self._read_fat_entry(c)
    return chain
```

#### 4.4 目录解析

```python
def _parse_dir(self, data):
    entries = []
    for i in range(0, len(data), 32):
        raw = data[i:i+32]
        first = raw[0]
        if first == 0: break
        if first == 0xE5: continue
        if raw[11] & 0x08: continue
        name = raw[0:8].decode('ascii', errors='replace')
        ext  = raw[8:11].decode('ascii', errors='replace')
        attr = raw[11]
        cluster = struct.unpack_from('<H', raw, 26)[0]
        size = struct.unpack_from('<I', raw, 28)[0]
        entries.append(FileEntry(name, ext, attr, cluster, size))
    return entries

def _build_tree(self):
    root_data = self._read_bytes(self.root_offset, self.root_sectors * self.bytes_per_sector)
    self.root = FileEntry("", "", ATTR_DIRECTORY, 0, 0)
    for e in self._parse_dir(root_data):
        self.root.children[e.display_name.upper()] = e
    self._parse_subdirs(self.root)

def _parse_subdirs(self, parent):
    for name, entry in list(parent.children.items()):
        if entry.is_directory and entry.name not in ('.', '..'):
            chain = self._get_cluster_chain(entry.cluster)
            data = bytearray()
            for c in chain:
                off = self.data_offset + (c - 2) * self.cluster_size
                data.extend(self._read_bytes(off, self.cluster_size))
            for e in self._parse_dir(bytes(data)):
                if e.name not in ('.', '..'):
                    entry.children[e.display_name.upper()] = e
            self._parse_subdirs(entry)
```

#### 4.5 查询接口

```python
def resolve_path(self, path):
    parts = [p for p in path.replace('\\', '/').split('/') if p]
    cur = self.root
    for p in parts:
        if p.upper() not in cur.children: return None
        cur = cur.children[p.upper()]
    return cur

def walk(self, path='/'):
    entries = self.list_dir(path) or []
    for e in entries:
        full = f"{path}/{e.display_name}".replace('//', '/')
        yield full, e
        if e.is_directory:
            yield from self.walk(full)

def list_dir(self, path='/'):
    e = self.resolve_path(path)
    return list(e.children.values()) if e and e.is_directory else None

def read_file(self, entry):
    chain = self._get_cluster_chain(entry.cluster)
    data = bytearray()
    for c in chain:
        off = self.data_offset + (c - 2) * self.cluster_size
        data.extend(self._read_bytes(off, self.cluster_size))
    return bytes(data[:entry.size])
```

#### 4.6 write_back_from_directory（核心）

```python
def write_back_from_directory(self, dir_path, save_path=None):
    max_cluster = self.total_sectors // self.sectors_per_cluster + 2
    fat = [0] * max_cluster
    if self.fat_type == 12:
        fat[0] = 0xF00 | (self.media_descriptor & 0xFF)
        fat[1] = 0xFFF
    else:
        fat[0] = 0xFF00 | (self.media_descriptor & 0xFF)
        fat[1] = 0xFFFF
    next_free = [2]

    fat_len = self.fat_sectors * self.bytes_per_sector
    root_len = self.root_sectors * self.bytes_per_sector
    data_len = (self.total_sectors - (self.reserved_sectors + self.num_fats * self.fat_sectors + self.root_sectors)) * self.bytes_per_sector

    self._write_bytes(self.fat_offset, b'\x00' * fat_len)
    for i in range(1, self.num_fats):
        self._write_bytes(self.fat_offset + i * fat_len, b'\x00' * fat_len)
    self._write_bytes(self.root_offset, b'\x00' * root_len)
    if data_len > 0:
        chunk = 65536
        written = 0
        while written < data_len:
            n = min(chunk, data_len - written)
            self._write_bytes(self.data_offset + written, b'\x00' * n)
            written += n

    SYSTEM_FILES = {'IO.SYS', 'MSDOS.SYS'}
    counters = {'files': 0, 'dirs': 0}

    def _process_dir(real_path, parent_cluster, is_root):
        entries = []
        used_names = set()
        items = sorted(os.listdir(real_path))
        if is_root:
            system = [n for n in items if n.upper() in SYSTEM_FILES]
            rest = [n for n in items if n.upper() not in SYSTEM_FILES]
            system.sort(key=lambda n: 0 if n.upper() == 'IO.SYS' else 1)
            items = system + sorted(rest)
        for item_name in items:
            item_path = os.path.join(real_path, item_name)
            name8, ext3 = _to_dos_name(item_name)
            name8, ext3 = _unique_83(name8, ext3, used_names)
            if os.path.isdir(item_path):
                dir_cluster = _alloc_cluster(fat, next_free)
                eoc = FAT12_EOC if self.fat_type == 12 else FAT16_EOC
                fat[dir_cluster] = eoc
                child_entries = _process_dir(item_path, dir_cluster, False)
                dir_data = bytearray()
                dir_data.extend(_make_entry(b'.       ', b'   ', ATTR_DIRECTORY, dir_cluster, 0))
                dir_data.extend(_make_entry(b'..      ', b'   ', ATTR_DIRECTORY, parent_cluster if not is_root else 0, 0))
                for ce in child_entries:
                    dir_data.extend(ce)
                remainder = len(dir_data) % self.cluster_size
                if remainder:
                    dir_data.extend(b'\x00' * (self.cluster_size - remainder))
                num_cl = len(dir_data) // self.cluster_size
                chain = [dir_cluster]
                for _ in range(num_cl - 1):
                    c = _alloc_cluster(fat, next_free)
                    fat[chain[-1]] = c
                    fat[c] = eoc
                    chain.append(c)
                for i, c in enumerate(chain):
                    chunk = dir_data[i * self.cluster_size:(i + 1) * self.cluster_size]
                    self._write_bytes(self.data_offset + (c - 2) * self.cluster_size, chunk)
                entries.append(_make_entry(name8, ext3, ATTR_DIRECTORY, dir_cluster, 0))
                counters['dirs'] += 1
            else:
                with open(item_path, 'rb') as fh:
                    file_data = fh.read()
                file_size = len(file_data)
                if file_size > 0:
                    num_cl = (file_size + self.cluster_size - 1) // self.cluster_size
                    chain = []
                    for _ in range(num_cl):
                        c = _alloc_cluster(fat, next_free)
                        if chain:
                            fat[chain[-1]] = c
                        chain.append(c)
                    eoc = FAT12_EOC if self.fat_type == 12 else FAT16_EOC
                    fat[chain[-1]] = eoc
                    for i, c in enumerate(chain):
                        chunk = file_data[i * self.cluster_size:(i + 1) * self.cluster_size]
                        self._write_bytes(self.data_offset + (c - 2) * self.cluster_size, chunk)
                    first_cluster = chain[0]
                else:
                    first_cluster = 0
                entries.append(_make_entry(name8, ext3, ATTR_ARCHIVE, first_cluster, file_size))
                counters['files'] += 1
        return entries

    root_entries = _process_dir(dir_path, 0, True)
    root_data = b''.join(root_entries)
    self._write_bytes(self.root_offset, root_data)

    fat_data = _build_fat_bytes(fat, self.fat_type, self.fat_sectors * self.bytes_per_sector)
    for i in range(self.num_fats):
        self._write_bytes(self.fat_offset + i * fat_len, fat_data)

    self.img.save(save_path)
    return counters['files'], counters['dirs']
```

辅助函数：

```python
def _to_dos_name(name):
    name = name.upper()
    if '.' in name:
        base, ext = name.rsplit('.', 1)
    else:
        base, ext = name, ''
    base = base[:8].ljust(8, ' ')
    ext = ext[:3].ljust(3, ' ')
    return base.encode('ascii', errors='replace'), ext.encode('ascii', errors='replace')

def _unique_83(name8, ext3, used):
    key = name8 + ext3
    if key not in used:
        used.add(key)
        return name8, ext3
    base = name8.rstrip(b' ')
    for n in range(1, 1000):
        suffix = f"~{n}".encode('ascii')
        mangled = base[:8 - len(suffix)] + suffix
        mangled = mangled.ljust(8)
        key = mangled + ext3
        if key not in used:
            used.add(key)
            return mangled, ext3
    raise RuntimeError("Cannot generate unique 8.3 name")

def _make_entry(name8, ext3, attr, cluster, size):
    e = bytearray(32)
    e[0:8] = name8[:8]
    e[8:11] = ext3[:3]
    e[11] = attr
    struct.pack_into('<H', e, 22, 0)
    struct.pack_into('<H', e, 24, 0x0021)
    struct.pack_into('<H', e, 26, cluster & 0xFFFF)
    struct.pack_into('<I', e, 28, size & 0xFFFFFFFF)
    return bytes(e)

def _alloc_cluster(fat, next_free):
    while next_free[0] < len(fat):
        if fat[next_free[0]] == 0:
            c = next_free[0]
            next_free[0] += 1
            return c
        next_free[0] += 1
    raise RuntimeError("Disk full")

def _build_fat_bytes(fat, fat_type, total_len):
    buf = bytearray(total_len)
    if fat_type == 12:
        for i, val in enumerate(fat):
            offset = i + (i // 2)
            if offset + 1 >= len(buf): break
            word = struct.unpack_from('<H', buf, offset)[0]
            if i & 1:
                word = (word & 0x000F) | ((val & 0x0FFF) << 4)
            else:
                word = (word & 0xF000) | (val & 0x0FFF)
            struct.pack_into('<H', buf, offset, word)
    else:
        for i, val in enumerate(fat):
            offset = i * 2
            if offset + 1 >= len(buf): break
            struct.pack_into('<H', buf, offset, val & 0xFFFF)
    return bytes(buf)
```

#### 4.7 字节级读写

```python
def _read_bytes(self, abs_offset, length):
    start_lba = abs_offset // self.img.sector_size
    count = (length + self.img.sector_size - 1) // self.img.sector_size
    data = self.img.read_sectors(start_lba, count)
    local_off = abs_offset % self.img.sector_size
    return data[local_off:local_off + length]

def _write_bytes(self, abs_offset, data):
    pos = 0
    ss = self.img.sector_size
    while pos < len(data):
        lba = (abs_offset + pos) // ss
        sec_off = (abs_offset + pos) % ss
        can_write = min(ss - sec_off, len(data) - pos)
        if sec_off == 0 and can_write == ss:
            self.img.write_sector(lba, data[pos:pos + ss])
        else:
            sec = bytearray(self.img.read_sector(lba))
            sec[sec_off:sec_off + can_write] = data[pos:pos + can_write]
            self.img.write_sector(lba, bytes(sec))
        pos += can_write
```

`temp_dir` 构建逻辑：

```python
def build_temp_dir(game):
    tmp = tempfile.mkdtemp(prefix='naiz_')
    for root, dirs, files in os.walk(REF_DISK):
        for f in files:
            src = os.path.join(root, f)
            rel = os.path.relpath(src, REF_DISK)
            dst = os.path.join(tmp, rel)
            os.makedirs(os.path.dirname(dst), exist_ok=True)
            shutil.copy2(src, dst)
    shutil.copy2(os.path.join(REF_CONFIG, 'CONFIG.SYS'), os.path.join(tmp, 'CONFIG.SYS'))
    shutil.copy2(os.path.join(REF_CONFIG, 'AUTOEXEC.BAT'), os.path.join(tmp, 'AUTOEXEC.BAT'))
    game_dir = os.path.join(GAMES_DIR, game)
    for root, dirs, files in os.walk(game_dir):
        for f in files:
            src = os.path.join(root, f)
            rel = os.path.relpath(src, GAMES_DIR)
            dst = os.path.join(tmp, rel)
            os.makedirs(os.path.dirname(dst), exist_ok=True)
            shutil.copy2(src, dst)
    return tmp
```

## 潜在风险

| 风险 | 影响 | 缓解 |
|------|------|------|
| `_write_bytes` 部分扇区 bug | HDI 损坏 | roundtrip 测试对比原文件 |
| IO.SYS 排序错误 | 不启动 | write_back 内 SYSTEM_FILES 排序必须在根目录生效 |
| GDC 初始化方向相反 | 无图形输出 | 先回文本模式打调试信息 |
| NP2kai 配置不对 | 黑屏 | 确认 HDD1FILE 路径 |
